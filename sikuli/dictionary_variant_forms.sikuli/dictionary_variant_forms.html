import a_setup
from test_helper import TestHelper

helper = TestHelper("help_about")

# Opening
#############

# Filter for "agbas"
helper.Click(, "'Lexicon Edit' not found", time=10)
helper.Click(Pattern().targetOffset(-52,23), "'Lexeme Form' column not found")
helper.Click(, "'Filter for' button not found")
paste("agbas")
type(Key.ENTER)
wait(15)

# Open Configure menu
helper.Click(, "'Tools' menu not found")
helper.Click(, "'Configure' menu not found")
helper.Click(Pattern().similar(0.95), "'Dictionary...' menu not found")

# Switch to Stem-based
if not exists():
    helper.Click(Pattern().targetOffset(-54,0), "Drop down menu next to 'Manage Views' not found")
    helper.Click(, "'Stem-based' not found")

# Uncheck all check boxes
main_entry = helper.Find(, "'Main Entry' not found")
click(main_entry)

# Get the region under and to the right of Main Entry
under = main_entry.below()
rightt = main_entry.right()
myregion = Region(under.getX(), under.getY(),
    under.getW() + rightt.getW(), under.getH())

while not myregion.exists():
    while myregion.exists(Pattern().similar(0.85)):
        myregion.click(Pattern().similar(0.85))
    type(Key.PAGE_DOWN)
while myregion.exists(Pattern().similar(0.85)):
        myregion.click(Pattern().similar(0.85))

# Check the Variant Forms boxes
while not exists():
    type(Key.PAGE_UP)
click()
variant_forms = helper.Find(, "'Variant Forms' not found")
focus_region = variant_forms

# This assumes all the "Variant Forms" options (expanded), and
# "Etymology" (collapsed) are on the same screen, with no scrolling.
while not focus_region.exists():
    # Check box if unchecked
    if not focus_region.exists(Pattern().similar(0.85)):
        type(Key.SPACE)
    # Move focus region down 16 px
    focus_region = focus_region.offset(Location(0, 16))
    type(Key.DOWN)

helper.Click(Pattern().similar(0.95), "'OK' button not found")

# Go to Dictionary view and wait (up to 30 seconds) for it to load
helper.Click(Pattern().similar(0.90), "'Dictionary' not found")
helper.Wait(, 30, "Dictionary loading timed out")

# Goal
###############
helper.Exists(, "Variant Forms dictionary view incorrect")
helper.write_success()
